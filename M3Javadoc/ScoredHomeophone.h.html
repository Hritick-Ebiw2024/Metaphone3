#include <string>
using namespace std;

/**
 * The scoredHomeophone class is a data class designed to wrap a
 * word which is a homeophone of the initial search string, its edit
 * distance (Levenshtein distance) score from this initial search
 * string, and a definition, if any.
 *
 */
class scoredHomeophone
{
private:

    /** Levenshtein distance of this word from the initial search string. */
    int score;

    /** Hypothetically, a homeophone of the initial search string. A homeophone
    * is a word which sounds similar, but not necessarily exactly the same,
    * as another word. */
    string word;

    /** A definition of the word, if WordNet has supplied one. */
    string definition;

public:

    /**
     * Parameterized constructor initialized with Levenshtein
     * distance score, and the word in question.
     *
     * @param inScore The Levenshtein distance score from the
     *                initial search string.
     * @param inWord  A character pointer to the homeophone of
     *                the original search string.
     */
    scoredHomeophone(int inScore, char* inWord)
    {
        score = inScore;
        word = inWord;
    }

    /**
     * Comparison operator required for objects which are to be stored
     * in sorted order in collection objects. This class is sorted by
     * Levenshtein distance score.
     *
     * @param
     *
     * @return true if the Levenshtein distance of the object on
     *         the right hand side of the operator is less than
     *         the Levenshtein distance of the object on the left.
     */
    bool scoredHomeophone::operator<( const scoredHomeophone& rhs)const
    {
        return (this->score < rhs.score);
    }   

    /**
     * Retrieve the word as a pointer to a character string.
     *
     * @return a character pointer to the word stored in this object.
     */
    const char* getWord()
    {
        return word.c_str();
    }

    /**
     * Retrieve the word as a std::string object.
     *
     * @return a reference to the string object containing the word stored in this object.
     */
    string getString()
    {       
        return word;
    }
    
    /**
     * Retrieve the Levenshtein distance score.
     *
     * @return an integer representing the Levenshtein distance
     *         between this word and the search string. 
     */
    int getScore()
    {
        return score;
    }
    
    /**
     * Set the Levenshtein distance score.
     *
     * @param an integer representing the Levenshtein distance
     *        between this word and the search string. 
     */
    void setScore(int inScore)
    {
        score = inScore;
    }

    /**
     * Store the definition of the word.
     *
     * @param a character pointer to the definition of the word.
     */
    void setDefinition(char* inDefinition)
    {
        definition = inDefinition ;
    }

    /**
     * Retrieve a pointer to the character string
     * storing the definition of the word.
     *
     * @return a character pointer to the definition of the word in question.
     *         if WordNet has not supplied a definition, a null string will be returned.
     */
    const char* getDefinition()
    {
        return definition.c_str();
    }   
};
